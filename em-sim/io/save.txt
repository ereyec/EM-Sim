//Two types of files we want to save, a grid workspace and a simulation screenshot

//Save a simulation screenshot
int saveSimOutputEx(simulationParams& sp, simulationState& ss){
//for simplicity, we will work with just the xz plane (y= 0), and just print the Ex component

std::string filename = "Sim-Output-Ex-step" + ss.step + ".csv";
FILE* outputFile;
outputFile = fopen(filename.c_str(), "w");

if(outputFile == NULL){
	return -1;
}

for(int z = 0; z < sp.z; z++){
for(int x = 0; x <= sp.x; x++){
	fprintf(outputFile, "%f, ", ss.yeeGrid(x, 0, z).Ex.current);
}
fprintf(outputFile, "%f\n", ss.yeeGrid(sp.x-1, 0, z).Ex.current);
}

fclose(outputFile);
return 0;
}
